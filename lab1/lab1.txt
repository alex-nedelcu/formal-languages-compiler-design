************** p2) Compute largest divisor of a natural number that is less than the number itself

	start

		define integer number;
		define integer divisor;

		read(number);

		loop (iterator=divisor, start=number-1, step=-1, until=1) {
			if (number % divisor == 0) {
				stop;
			}
		}

		display("divisor-is:");
		display(divisor);

	end
